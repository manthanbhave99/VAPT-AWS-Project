import json
import boto3
import time

client = boto3.client('lambda')

def lambda_handler(event, context):

	response = client.invoke(
		FunctionName='subfinder',
		InvocationType='RequestResponse',
		)

	responseJson = json.load(response['Payload'])["subs"]
	subdomains = responseJson.split("\\n")

	for sd in subdomains:
		sub = sd.strip("\"")
		httpx_test(sub)


def httpx_test(subdomain):

    # boto3 client
    client = boto3.client("ec2")
    ssm = boto3.client("ssm")

    # getting instance information
    describeInstance = client.describe_instances()

    InstanceId = []
    # fetching instance id of the running instances
    for i in describeInstance["Reservations"]:
        for instance in i["Instances"]:
            if instance["State"]["Name"] == "running":
                InstanceId.append(instance["InstanceId"])

    # looping through instance ids
    for instanceid in InstanceId:
        # command to be executed on instance
        response = ssm.send_command(
            InstanceIds=[instanceid],
            DocumentName="AWS-RunShellScript",
            Parameters={
                "commands": ["echo " + subdomain + " | httpx -t 5"]
            },
        )

        # fetching command id for the output
        command_id = response["Command"]["CommandId"]

        time.sleep(4)

        # fetching command output
        output = ssm.get_command_invocation(CommandId=command_id, InstanceId=instanceid)
        print()

    return {"statusCode": 200, "httpx": json.dumps(output["StandardOutputContent"])}
